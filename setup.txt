setup script
	apt-get install apache2 postgresql postgresql-client python-pip git apache2-dev python3-dev libapache2-mod-wsgi-py3 nginx python3 python3-dev
	
	// DATABASE
	su - postgres
	psql
	CREATE DATABASE lifesaving_rankings;
	CREATE USER lifesaving_rankings_django PASSWORD '5EydU!G3jwTuewDq';
	GRANT ALL ON DATABASE lifesaving_rankings TO lifesaving_rankings_django;
	//GRANT ALL ON ALL TABLES IN SCHEMA lifesaving_rankings TO lifesaving_rankings_django;
	\q
	
	// INSTALL PROJECT
	pip install virtualenv
	virtualenv -p python3 /var/www/lifesaving_rankings
	source /var/www/envs/lifesaving_rankings/bin/activate
	cd /var/www/
	git clone https://rubenvanerk@bitbucket.org/rubenvanerk/lifesaving-rankings.git
	chown -R www-data:www-data /var/www/lifesaving-rankings
	pip install -r /var/www/lifesaving-rankings/requirements.txt
	
	// CONFIGURE APACHE
	/etc/init.d/apache2 restart
		add to /etc/apache2/sites-available/lifesaving-rankings.conf:		
			<VirtualHost *:80>
				WSGIScriptAlias / /var/www/lifesaving-rankings/lifesaving_rankings/wsgi.py
				ServerAdmin ruben@rubenvanerk.nl
				ServerName lifesavingrankings.nl

				<Directory /var/www/lifesaving-rankings/lifesaving_rankings>
						<Files wsgi.py>
								Require all granted
						</Files>
				</Directory>

				WSGIDaemonProcess lifesavingrankings.nl python-home=/var/www/envs/lifesaving_rankings python-path=/var/www/lifesaving-rankings
				WSGIProcessGroup lifesavingrankings.nl
			</VirtualHost>
		add to /etc/apache2/apache2.conf
			WSGIPythonHome /var/www/envs/lifesaving_rankings
			WSGIPythonPath /var/www/lifesaving-rankings
			
	//CONFIGURE NGINX
	/etc/init.d/apache2 stop
	apt-get install nginx-full
	/etc/init.d/apache2 start
		add to /etc/nginx/sites-available/lifesaving-rankings.conf
		server {
			listen   8000;
			server_name  localhost;

			 access_log  /var/log/nginx/aa8000.access.log;
			 error_log  /var/log/nginx/aa8000.error.log;

			   location / {
				   index  index.html index.htm;
			   }

			 location /static {
				autoindex on;
				alias   /var/www/lifesaving-rankings/lifesaving_rankings/;
			 }
		}
	ln -s /etc/nginx/sites-available/lifesaving-rankings.conf /etc/nginx/sites-enabled/
	rm /etc/nginx/sites-enabled/default
	service nginx reload	
	
	python /var/www/lifesaving-rankings/manage.py migrate
